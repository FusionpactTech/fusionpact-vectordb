quickstart.js
const { FusionEngine, RAGPipeline } = require('fusionpact');

// Create engine & RAG pipeline
const engine = new FusionEngine();
const rag = new RAGPipeline(engine, { embedder: 'ollama' });

// Ingest any text ‚Äî auto chunks, embeds, indexes
await rag.ingest('Your document text...', {
  source: 'safety-manual.pdf'
});

// Search with natural language
const ctx = await rag.buildContext(
  'What are the safety protocols?'
);

console.log(ctx.prompt); // ‚Üí LLM-ready prompt
